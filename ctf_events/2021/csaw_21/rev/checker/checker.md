## Challenge Description

> What's up with all the zeros and ones? Where are my letters and numbers?

## Solution

```py
def up(x):
    x = [f"{ord(x[i]) << 1:08b}" for i in range(len(x))]
    return ''.join(x)

def down(x):
    x = ''.join(['1' if x[i] == '0' else '0' for i in range(len(x))])
    return x

def right(x, d):
    x = x[d:] + x[0:d]
    return x

def left(x, d):
    x = right(x, len(x) - d)
    return x[::-1]

def encode(plain):
    d = 24
    print('init : 01100001011000100110001101100100')
    x = up(plain)
    print('up   : ' + x)
    x = right(x, d)
    print('right: ' + x)
    x = down(x)
    print('down : ' + x)
    x = left(x, d)
    print('left : ' + x)
    return x

def decode(encoded):
    d = 24
    x = right(encoded[::-1], d)
    x = down(x)
    x = right(x, len(encoded) - d)
    return x
```
